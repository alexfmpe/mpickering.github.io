   29 -- | Transform concrete annotations into relative annotations which are
   30 -- more useful when transforming an AST.
   31 relativiseApiAnns :: Annotate ast
   32                   => GHC.Located ast
   33                   -> GHC.ApiAnns
   34                   -> Anns
   35 relativiseApiAnns modu@(GHC.L ss _) ghcAnns
   36    = runDelta (markLocated modu) ghcAnns ss

   40 data DeltaState = DeltaState
   41              { priorEndPosition :: GHC.SrcSpan
   42                -- | Ordered list of comments still to be allocated
   43              , apComments :: [Comment]
   44                -- | The original GHC Delta Annotations
   45              , apAnns :: GHC.ApiAnns
   46              }
   47 
   48 data DeltaStack = DeltaStack
   49                { -- | Current `SrcSpan`
   50                  curSrcSpan :: GHC.SrcSpan
   51                  -- | Constuctor of current AST element, useful for
   52                  -- debugging
   53                , annConName :: AnnConName
   54                  -- | Start column of the current layout block
   55                , layoutStart :: Int
   56                }
   57 
   58 initialDeltaStack :: DeltaStack
   59 initialDeltaStack =
   60   DeltaStack
   61     { curSrcSpan = GHC.noSrcSpan
   62     , annConName = annGetConstr ()
   63     , layoutStart = 0
   64     }
   65 
   66 defaultDeltaState :: GHC.SrcSpan -> GHC.ApiAnns -> DeltaState
   67 defaultDeltaState priorEnd ga =
   68   let cs = flattenedComments ga in
   69     DeltaState
   70       { priorEndPosition = priorEnd
   71       , apComments = cs
   72       , apAnns     = ga    -- $
   73       }

   75 data DeltaWriter = DeltaWriter
   76   { -- Final list of annotations
   77     finalAnns :: Endo (Map.Map AnnKey Annotation)
   78     -- Used locally to pass Keywords, delta pairs relevant to a specific
   79     -- subtree to the parent.
   80   , annKds :: [(KeywordId, DeltaPos)]
   81     -- Used locally to report a subtrees aderhence to haskell's layout
   82     -- rules.
   83   , propOffset :: First Int -- Used to pass the offset upwards
   84   }
   85 
   86 
   87 tellFinalAnn :: (AnnKey, Annotation) -> Delta ()
   88 tellFinalAnn (k, v) =
   89   tell (mempty { finalAnns = Endo (Map.insertWith (<>) k v) })
   90 
   91 tellKd :: (KeywordId, DeltaPos) -> Delta ()
   92 tellKd kd = tell (mempty { annKds = [kd] })
   93 
   94 setLayoutFlag :: GHC.AnnKeywordId -> Delta () -> Delta ()
   95 setLayoutFlag kwid action = do
   96   c <-  srcSpanStartColumn . head <$> getAnnotationDelta kwid
   97   tell (mempty { propOffset = First (Just c) })
   98   local (\s -> s { layoutStart = c }) action
   99 
  100 instance Monoid DeltaWriter where
  101   mempty = DeltaWriter mempty mempty mempty
  102   (DeltaWriter a b e) `mappend` (DeltaWriter c d f) = DeltaWriter (a <> c) (b <> d) (e <> f)
  103

  105 -- | Type used in the Delta Monad. The state variables maintain
  106 --    - the current SrcSpan and the constructor of the thing it encloses
  107 --      as a stack to the root of the AST as it is traversed,
  108 --    - the srcspan of the last thing annotated, to calculate delta's from
  109 --    - extra data needing to be stored in the monad
  110 --    - the annotations provided by GHC
  111 type Delta a = RWS DeltaStack DeltaWriter DeltaState a
  112 
  113 runDelta :: Annotated () -> GHC.ApiAnns -> GHC.SrcSpan -> Anns
  114 runDelta action ga priorEnd =
  115   ($ mempty) . appEndo . finalAnns . snd
  116   . (\next -> execRWS next initialDeltaStack (defaultDeltaState priorEnd ga))
  117   . simpleInterpret $ action
  118 
  119 
  120 simpleInterpret :: Annotated a -> Delta a
  121 simpleInterpret = iterTM go
  122   where
  123     go :: AnnotationF (Delta a) -> Delta a
  124     go (MarkEOF next) = addEofAnnotation >> next
  125     go (MarkPrim kwid _ next) =
  126       addDeltaAnnotation kwid >> next
  127     go (MarkOutside akwid kwid next) =
  128       addDeltaAnnotationsOutside akwid kwid >> next
  129     go (MarkInside akwid next) =
  130       addDeltaAnnotationsInside akwid >> next
  131     go (MarkMany akwid next) = addDeltaAnnotations akwid >> next
  132     go (MarkOffsetPrim akwid n _ next) = addDeltaAnnotationLs akwid n >> next
  133     go (MarkAfter akwid next) = addDeltaAnnotationAfter akwid >> next
  134     go (WithAST lss layoutflag prog next) =
  135       withAST lss layoutflag (simpleInterpret prog) >> next
  136     go (OutputKD (kwid, (_, dp)) next) = tellKd (dp, kwid) >> next
  137     go (CountAnns kwid next) = countAnnsDelta kwid >>= next
  138     go (SetLayoutFlag kwid action next) = setLayoutFlag kwid (simpleInterpret action)  >> next
  139     go (MarkExternal ss akwid _ next) = addDeltaAnnotationExt ss akwid >> next

  144 flattenedComments :: GHC.ApiAnns -> [Comment]
  145 flattenedComments (_,cm) = map tokComment . GHC.sortLocated . concat $ Map.elems cm

  149 getSrcSpanDelta :: Delta GHC.SrcSpan
  150 getSrcSpanDelta = asks curSrcSpan
  151 
  152 withSrcSpanDelta :: Data a => (GHC.Located a) -> Delta b -> Delta b
  153 withSrcSpanDelta (GHC.L l a) =
  154   local (\s -> s { curSrcSpan = l
  155                  , annConName = annGetConstr a
  156                  })

  158 getUnallocatedComments :: Delta [Comment]
  159 getUnallocatedComments = gets apComments

  161 putUnallocatedComments :: [Comment] -> Delta ()
  162 putUnallocatedComments cs = modify (\s -> s { apComments = cs } )

  166 adjustDeltaForOffsetM :: DeltaPos -> Delta DeltaPos
  167 adjustDeltaForOffsetM dp = do
  168   colOffset <- asks layoutStart
  169   return (adjustDeltaForOffset colOffset dp)
  170 
  171 adjustDeltaForOffset :: Int -> DeltaPos -> DeltaPos
  172 adjustDeltaForOffset _colOffset dp@(DP (0,_)) = dp -- same line
  173 adjustDeltaForOffset  colOffset    (DP (l,c)) = DP (l,c - colOffset)

  181 setPriorEnd pe = modify (\s -> s { priorEndPosition = pe })

  185 getAnnotationDelta :: GHC.AnnKeywordId -> Delta [GHC.SrcSpan]
  186 getAnnotationDelta an = do
  187     ga <- gets apAnns
  188     ss <- getSrcSpanDelta
  189     return $ GHC.getAnnotation ga ss an
  190 
  191 getAndRemoveAnnotationDelta :: GHC.SrcSpan -> GHC.AnnKeywordId -> Delta [GHC.SrcSpan]
  192 getAndRemoveAnnotationDelta sp an = do
  193     ga <- gets apAnns
  194     let (r,ga') = GHC.getAndRemoveAnnotation ga sp an
  195     r <$ modify (\s -> s { apAnns = ga' })

  199 tokComment t@(GHC.L lt _) = Comment (ss2span lt) (ghcCommentText t)

  203 -- |Add some annotation to the currently active SrcSpan
  204 addAnnotationsDelta :: Annotation -> Delta ()
  205 addAnnotationsDelta ann = do
  206     l <- ask
  207     tellFinalAnn ((getAnnKey l),ann)

  209 getAnnKey :: DeltaStack -> AnnKey
  210 getAnnKey DeltaStack {curSrcSpan, annConName} = AnnKey curSrcSpan annConName

  219 setLayoutOffset :: Int -> Delta a -> Delta a
  220 setLayoutOffset lhs = local (\s -> s { layoutStart = lhs })

  222 -- | Enter a new AST element. Maintain SrcSpan stack
  223 withAST :: Data a => GHC.Located a -> LayoutFlag -> Delta b -> Delta b
  224 withAST lss@(GHC.L ss _) layout action = do
  225   -- Calculate offset required to get to the start of the SrcSPan
  226   pe <- getPriorEnd
  227   off <- asks layoutStart
  228   let whenLayout = case layout of
  229                         LayoutRules -> setLayoutOffset (srcSpanStartColumn ss)
  230                         NoLayoutRules -> id
  231   (whenLayout .  withSrcSpanDelta lss) (do
  232 
  233     let maskWriter s = s { annKds = []
  234                          , propOffset = First Nothing }
  235 
  236     (res, w) <-
  237       (censor maskWriter (listen action))
  238     let edp = adjustDeltaForOffset (fromMaybe off (getFirst $ propOffset w))
  239                 (deltaFromSrcSpans pe ss)
  240 --    let (dp) = getCurrentDP layout  ss prior
  241     let kds = annKds w
  242     addAnnotationsDelta (Ann edp (srcSpanStartColumn ss - off) kds)
  243     --  `debug` ("leaveAST:(ss,finaledp,dp,nl,kds)=" ++ show (showGhc ss,edp,dp,nl,kds))
  244     return res)

  250 getCurrentDP :: LayoutFlag -> GHC.SrcSpan -> GHC.SrcSpan -> (ColOffset,LineChanged)
  251 getCurrentDP layoutOn ss ps =
  252   -- Note: the current col offsets are not needed here, any
  253   -- indentation should be fully nested in an AST element
  254   let
  255       colOffset = if srcSpanStartLine ss == srcSpanStartLine ps
  256                     then srcSpanStartColumn ss - srcSpanStartColumn ps
  257                     else srcSpanStartColumn ss
  258 
  259   in colOffset
  260     `debug` ("getCurrentDP:(layoutOn=" ++ show layoutOn)

  268 -- |Split the ordered list of comments into ones that occur prior to
  269 -- the given SrcSpan and the rest
  270 allocatePriorComments :: [Comment] -> GHC.SrcSpan -> ([Comment],[Comment])
  271 allocatePriorComments cs ss = partition isPrior cs
  272   where
  273     (start,_) = ss2span ss
  274     isPrior (Comment s _)  = (fst s) < start
  275       `debug` ("allocatePriorComments:(s,ss,cond)=" ++ showGhc (s,ss,(fst s) < start))

  279 addAnnotationWorker :: KeywordId -> GHC.SrcSpan -> Delta ()
  280 addAnnotationWorker ann pa = do
  281   if not (isPointSrcSpan pa)
  282     then do
  283       pe <- getPriorEnd
  284       ss <- getSrcSpanDelta
  285       let p = deltaFromSrcSpans pe pa
  286       case (ann,isGoodDelta p) of
  287         (G GHC.AnnComma,False) -> return ()
  288         (G GHC.AnnSemi, False) -> return ()
  289         (G GHC.AnnOpen, False) -> return ()
  290         (G GHC.AnnClose,False) -> return ()
  291         _ -> do
  292           cs <- getUnallocatedComments
  293           let (allocated,cs') = allocatePriorComments cs pa
  294           putUnallocatedComments cs'
  295           return () `debug`("addAnnotationWorker:(ss,pa,allocated,cs)=" ++ showGhc (ss,pa,allocated,cs))
  296           mapM_ addDeltaComment allocated
  297           p' <- adjustDeltaForOffsetM p
  298           addAnnDeltaPos (ss,ann) p'
  299           setPriorEnd pa
  300               -- `debug` ("addDeltaAnnotationWorker:(ss,pe,pa,p,ann)=" ++ show (ss2span ss,ss2span pe,ss2span pa,p,ann))
  301     else do
  302       return ()
  303           -- `debug` ("addDeltaAnnotationWorker::point span:(ss,ma,ann)=" ++ show (ss2span pa,ann))
  304

  307 addDeltaComment :: Comment -> Delta ()
  308 addDeltaComment (Comment paspan str) = do
  309   let pa = span2ss paspan
  310   pe <- getPriorEnd
  311   ss <- getSrcSpanDelta
  312   let p = deltaFromSrcSpans pe pa
  313   p' <- adjustDeltaForOffsetM p
  314   setPriorEnd pa
  315   let e = ss2deltaP (ss2posEnd pe) (snd paspan)
  316   e' <- adjustDeltaForOffsetM e
  317   addAnnDeltaPos (ss,AnnComment (DComment (p',e') str)) p'

  321 -- | Look up and add a Delta annotation at the current position, and
  322 -- advance the position to the end of the annotation
  323 addDeltaAnnotation :: GHC.AnnKeywordId -> Delta ()
  324 addDeltaAnnotation ann = do
  325   ss <- getSrcSpanDelta
  326   when (ann == GHC.AnnVal) (debugM (showGhc ss))
  327   ma <- getAnnotationDelta ann
  328   when (ann == GHC.AnnVal && null ma) (debugM "empty")
  329   case nub ma of -- ++AZ++ TODO: get rid of duplicates earlier
  330     [] -> return () `debug` ("addDeltaAnnotation empty ma for:" ++ show ann)
  331     [pa] -> addAnnotationWorker (G ann) pa
  332     _ -> error $ "addDeltaAnnotation:(ss,ann,ma)=" ++ showGhc (ss,ann,ma)
  333 
  334 -- | Look up and add a Delta annotation appearing beyond the current
  335 -- SrcSpan at the current position, and advance the position to the
  336 -- end of the annotation
  337 addDeltaAnnotationAfter :: GHC.AnnKeywordId -> Delta ()
  338 addDeltaAnnotationAfter ann = do
  339   ss <- getSrcSpanDelta
  340   ma <- getAnnotationDelta ann
  341   let ma' = filter (\s -> not (GHC.isSubspanOf s ss)) ma
  342   case ma' of
  343     [] -> return () `debug` ("addDeltaAnnotation empty ma")
  344     [pa] -> addAnnotationWorker (G ann) pa
  345     _ -> error $ "addDeltaAnnotation:(ss,ann,ma)=" ++ showGhc (ss,ann,ma)
  346 
  347 -- | Look up and add a Delta annotation at the current position, and
  348 -- advance the position to the end of the annotation
  349 addDeltaAnnotationLs :: GHC.AnnKeywordId -> Int -> Delta ()
  350 addDeltaAnnotationLs ann off = do
  351   ma <- getAnnotationDelta ann
  352   case (drop off ma) of
  353     [] -> return ()
  354         -- `debug` ("addDeltaAnnotationLs:missed:(off,pe,ann,ma)=" ++ show (off,ss2span pe,ann,fmap ss2span ma))
  355     (pa:_) -> addAnnotationWorker (G ann) pa
  356 
  357 -- | Look up and add possibly multiple Delta annotation at the current
  358 -- position, and advance the position to the end of the annotations
  359 addDeltaAnnotations :: GHC.AnnKeywordId -> Delta ()
  360 addDeltaAnnotations ann = do
  361   ma <- getAnnotationDelta ann
  362   let do_one ap' = addAnnotationWorker (G ann) ap'
  363                     -- `debug` ("addDeltaAnnotations:do_one:(ap',ann)=" ++ showGhc (ap',ann))
  364   mapM_ do_one (sort ma)
  365 
  366 -- | Look up and add possibly multiple Delta annotations enclosed by
  367 -- the current SrcSpan at the current position, and advance the
  368 -- position to the end of the annotations
  369 addDeltaAnnotationsInside :: GHC.AnnKeywordId -> Delta ()
  370 addDeltaAnnotationsInside ann = do
  371   ss <- getSrcSpanDelta
  372   ma <- getAnnotationDelta ann
  373   let do_one ap' = addAnnotationWorker (G ann) ap'
  374                     -- `debug` ("addDeltaAnnotations:do_one:(ap',ann)=" ++ showGhc (ap',ann))
  375   let filtered = (sort $ filter (\s -> GHC.isSubspanOf s ss) ma)
  376   mapM_ do_one filtered
  377 
  378 -- | Look up and add possibly multiple Delta annotations not enclosed by
  379 -- the current SrcSpan at the current position, and advance the
  380 -- position to the end of the annotations
  381 addDeltaAnnotationsOutside :: GHC.AnnKeywordId -> KeywordId -> Delta ()
  382 addDeltaAnnotationsOutside gann ann = do
  383   ss <- getSrcSpanDelta
  384   if ss2span ss == ((1,1),(1,1))
  385     then return ()
  386     else do
  387       -- ma <- getAnnotationDelta ss gann
  388       ma <- getAndRemoveAnnotationDelta ss gann
  389       let do_one ap' = addAnnotationWorker ann ap'
  390       mapM_ do_one (sort $ filter (\s -> not (GHC.isSubspanOf s ss)) ma)
  391 
  392 -- | Add a Delta annotation at the current position, and advance the
  393 -- position to the end of the annotation
  394 addDeltaAnnotationExt :: GHC.SrcSpan -> GHC.AnnKeywordId -> Delta ()
  395 addDeltaAnnotationExt s ann = do
  396   addAnnotationWorker (G ann) s

  399 addEofAnnotation :: Delta ()
  400 addEofAnnotation = do
  401   pe <- getPriorEnd
  402   ss <- getSrcSpanDelta
  403   ma <- withSrcSpanDelta (GHC.noLoc ()) (getAnnotationDelta GHC.AnnEofPos)
  404   case ma of
  405     [] -> return ()
  406     (pa:pss) -> do
  407       cs <- getUnallocatedComments
  408       mapM_ addDeltaComment cs
  409       let DP (r,c) = deltaFromSrcSpans pe pa
  410       addAnnDeltaPos (ss,G GHC.AnnEofPos) (DP (r, c - 1))
  411       setPriorEnd pa `warn` ("Trailing annotations after Eof: " ++ showGhc pss)
  412 
  413 
  414 countAnnsDelta :: GHC.AnnKeywordId -> Delta Int
  415 countAnnsDelta ann = do
  416   ma <- getAnnotationDelta ann
  417   return (length ma)
