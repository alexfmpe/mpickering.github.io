   25 -- | Transform concrete annotations into relative annotations which are
   26 -- more useful when transforming an AST.
   27 relativiseApiAnns :: Annotate ast
   28                   => GHC.Located ast
   29                   -> GHC.ApiAnns
   30                   -> Anns
   31 relativiseApiAnns modu@(GHC.L ss _) ghcAnns
   32    = runDelta (markLocated modu) ghcAnns ss

   36 data DeltaState = DeltaState
   37              { priorEndPosition :: GHC.SrcSpan
   38                -- | Ordered list of comments still to be allocated
   39              , apComments :: [Comment]
   40                -- | The original GHC DeltaI Annotations
   41              , apAnns :: GHC.ApiAnns
   42              }
   43 
   44 data DeltaStack = DeltaStack
   45                { -- | Current `SrcSpan`
   46                  curSrcSpan :: GHC.SrcSpan
   47                  -- |  `SrcSpan` of the immediately prior scope
   48                , prevSrcSpan :: GHC.SrcSpan
   49                  -- | The offset required to get from the prior end point to the
   50                  -- | The offset required to get from the prior end point to the
   51                  -- start of the current SrcSpan. Accessed via `getEntryDP`
   52                , offset     :: DeltaPos
   53                  -- | Offsets for the elements annotated in this `SrcSpan`
   54                -- | Indicates whether the contents of this SrcSpan are
   55                -- subject to vertical alignment layout rules
   56                  -- | The constructor name of the AST element, to
   57                  -- distinguish between nested elements that have the same
   58                  -- `SrcSpan` in the AST.
   59                , annConName :: AnnConName
   60                }
   61 
   62 initialDeltaStack :: DeltaStack
   63 initialDeltaStack =
   64   DeltaStack
   65     { curSrcSpan = GHC.noSrcSpan
   66     , prevSrcSpan = GHC.noSrcSpan
   67     , offset = DP (0,0)
   68     , annConName = annGetConstr ()
   69     }
   70 
   71 defaultDeltaState :: GHC.SrcSpan -> GHC.ApiAnns -> DeltaState
   72 defaultDeltaState priorEnd ga =
   73   let cs = flattenedComments ga in
   74     DeltaState
   75       { priorEndPosition = priorEnd
   76       , apComments = cs
   77       , apAnns     = ga    -- $
   78       }

   80 data DeltaWriter = DeltaWriter
   81   { -- Final list of annotations
   82     finalAnns :: Endo (Map.Map AnnKey Annotation)
   83     -- Used locally to pass Keywords, delta pairs relevant to a specific
   84     -- subtree to the parent.
   85   , annKds :: [(KeywordId, DeltaPos)]
   86     -- Used locally to report a subtrees aderhence to haskell's layout
   87     -- rules.
   88   , layoutFlag :: LayoutFlag
   89   }
   90 
   91 
   92 tellFinalAnn :: (AnnKey, Annotation) -> Delta ()
   93 tellFinalAnn (k, v) =
   94   tell (mempty { finalAnns = Endo (Map.insertWith (<>) k v) })
   95 
   96 tellKd :: (KeywordId, DeltaPos) -> Delta ()
   97 tellKd kd = tell (mempty { annKds = [kd] })
   98 
   99 setLayoutFlag :: Delta ()
  100 setLayoutFlag = tell (mempty {layoutFlag = LayoutRules})
  101 
  102 instance Monoid DeltaWriter where
  103   mempty = DeltaWriter mempty mempty mempty
  104   (DeltaWriter a b e) `mappend` (DeltaWriter c d f) = DeltaWriter (a <> c) (b <> d) (e <> f)

  107 -- | Type used in the Delta Monad. The state variables maintain
  108 --    - the current SrcSpan and the constructor of the thing it encloses
  109 --      as a stack to the root of the AST as it is traversed,
  110 --    - the srcspan of the last thing annotated, to calculate delta's from
  111 --    - extra data needing to be stored in the monad
  112 --    - the annotations provided by GHC
  113 type Delta a = RWS DeltaStack DeltaWriter DeltaState a
  114 
  115 runDelta :: Annotated () -> GHC.ApiAnns -> GHC.SrcSpan -> Anns
  116 runDelta action ga priorEnd =
  117   ($ mempty) . appEndo . finalAnns . snd
  118   . (\next -> execRWS next initialDeltaStack (defaultDeltaState priorEnd ga))
  119   . simpleInterpret $ action
  120 
  121 
  122 simpleInterpret :: Annotated a -> Delta a
  123 simpleInterpret = iterTM go
  124   where
  125     go :: AnnotationF (Delta a) -> Delta a
  126     go (MarkEOF next) = addEofAnnotation >> next
  127     go (MarkPrim kwid _ next) =
  128       addDeltaAnnotation kwid >> next
  129     go (MarkOutside akwid kwid next) =
  130       addDeltaAnnotationsOutside akwid kwid >> next
  131     go (MarkInside akwid next) =
  132       addDeltaAnnotationsInside akwid >> next
  133     go (MarkMany akwid next) = addDeltaAnnotations akwid >> next
  134     go (MarkOffsetPrim akwid n _ next) = addDeltaAnnotationLs akwid n >> next
  135     go (MarkAfter akwid next) = addDeltaAnnotationAfter akwid >> next
  136     go (WithAST lss layoutflag prog next) =
  137       withAST lss layoutflag (simpleInterpret prog) >>= next
  138     go (OutputKD (kwid, (_, dp)) next) = tellKd (dp, kwid) >> next
  139     go (CountAnns kwid next) = countAnnsDelta kwid >>= next
  140     go (SetLayoutFlag next) = setLayoutFlag >> next
  141     go (MarkExternal ss akwid _ next) = addDeltaAnnotationExt ss akwid >> next

  147 flattenedComments (_,cm) = map tokComment . GHC.sortLocated . concat $ Map.elems cm

  151 getSrcSpanDelta :: Delta GHC.SrcSpan
  152 getSrcSpanDelta = asks curSrcSpan
  153 
  154 getPriorSrcSpanDelta :: Delta GHC.SrcSpan
  155 getPriorSrcSpanDelta = asks prevSrcSpan
  156 
  157 withSrcSpanDelta :: Data a => (GHC.Located a) -> DeltaPos -> Delta b -> Delta b
  158 withSrcSpanDelta (GHC.L l a) edp =
  159   local (\s -> let previousSrcSpan = curSrcSpan s in
  160                s { curSrcSpan = l
  161                  , prevSrcSpan = previousSrcSpan
  162                  , offset = edp
  163                  , annConName = annGetConstr a
  164                  })

  166 getEntryDP :: Delta DeltaPos
  167 getEntryDP = asks offset

  169 getUnallocatedComments :: Delta [Comment]
  170 getUnallocatedComments = gets apComments
  171 
  172 putUnallocatedComments :: [Comment] -> Delta ()
  173 putUnallocatedComments cs = modify (\s -> s { apComments = cs } )

  177 adjustDeltaForOffsetM :: DeltaPos -> Delta DeltaPos
  178 adjustDeltaForOffsetM dp = do
  179   colOffset <- getCurrentColOffset
  180   return (adjustDeltaForOffset colOffset dp)
  181 
  182 adjustDeltaForOffset :: Int -> DeltaPos -> DeltaPos
  183 adjustDeltaForOffset _colOffset dp@(DP (0,_)) = dp -- same line
  184 adjustDeltaForOffset  colOffset    (DP (l,c)) = DP (l,c - colOffset)
  185

  188 -- | Get the current column offset
  189 getCurrentColOffset :: Delta ColOffset
  190 getCurrentColOffset = srcSpanStartColumn <$> getSrcSpanDelta
  191 
  192 -- |Get the difference between the current and the previous
  193 -- colOffsets, if they are on the same line
  194 getCurrentDP :: LayoutFlag -> Delta (ColOffset,LineChanged)
  195 getCurrentDP layoutOn = do
  196   -- Note: the current col offsets are not needed here, any
  197   -- indentation should be fully nested in an AST element
  198   ss <- getSrcSpanDelta
  199   ps <- getPriorSrcSpanDelta
  200   let boolLayoutFlag = case layoutOn of { LayoutRules -> True; NoLayoutRules -> False}
  201       colOffset = if srcSpanStartLine ss == srcSpanStartLine ps
  202                     then srcSpanStartColumn ss - srcSpanStartColumn ps
  203                     else srcSpanStartColumn ss
  204       r = case ( boolLayoutFlag , srcSpanStartLine ss == srcSpanStartLine ps) of
  205              (True,  True) -> (colOffset, LayoutLineSame)
  206              (True, False) -> (colOffset, LayoutLineChanged)
  207              (False, True) -> (colOffset, LineSame)
  208              (False,False) -> (colOffset, LineChanged)
  209   return r
  210     `debug` ("getCurrentDP:layoutOn=" ++ show layoutOn)

  223 getAnnotationDelta :: GHC.AnnKeywordId -> Delta [GHC.SrcSpan]
  224 getAnnotationDelta an = do
  225     ga <- gets apAnns
  226     ss <- getSrcSpanDelta
  227     return $ GHC.getAnnotation ga ss an

  229 getAndRemoveAnnotationDelta :: GHC.SrcSpan -> GHC.AnnKeywordId -> Delta [GHC.SrcSpan]
  230 getAndRemoveAnnotationDelta sp an = do
  231     ga <- gets apAnns
  232     let (r,ga') = GHC.getAndRemoveAnnotation ga sp an
  233     r <$ modify (\s -> s { apAnns = ga' })

  241 -- |Add some annotation to the currently active SrcSpan
  242 addAnnotationsDelta :: Annotation -> Delta ()
  243 addAnnotationsDelta ann = do
  244     l <- ask

  257 -- | Enter a new AST element. Maintain SrcSpan stack
  258 withAST :: Data a => GHC.Located a -> LayoutFlag -> Delta b -> Delta b
  259 withAST lss layout action = do
  260   -- Calculate offset required to get to the start of the SrcSPan
  261   pe <- getPriorEnd
  262   let ss = (GHC.getLoc lss)
  263   let edp = deltaFromSrcSpans pe ss
  264   edp' <- adjustDeltaForOffsetM edp
  265   -- need to save edp', and put it in Annotation
  266 
  267   withSrcSpanDelta lss edp' (do
  268 
  269     let maskWriter s = s { annKds = []
  270                          , layoutFlag = NoLayoutRules }
  271 
  272     (res, w) <-
  273       (censor maskWriter (listen action))
  274 
  275     (dp,nl)  <- getCurrentDP (layout <> layoutFlag w)
  276     finaledp <- getEntryDP
  277     let kds = annKds w
  278     addAnnotationsDelta (Ann finaledp nl (srcSpanStartColumn ss) dp kds)
  279       `debug` ("leaveAST:(ss,edp,dp,kds)=" ++ show (showGhc ss,edp,dp,kds,dp))
  280     return res)

  289 -- |Split the ordered list of comments into ones that occur prior to
  290 -- the given SrcSpan and the rest
  291 allocatePriorComments :: [Comment] -> GHC.SrcSpan -> ([Comment],[Comment])
  292 allocatePriorComments cs ss = partition isPrior cs
  293   where
  294     (start,_) = ss2span ss
  295     isPrior (Comment s _)  = (fst s) < start
  296       `debug` ("allocatePriorComments:(s,ss,cond)=" ++ showGhc (s,ss,(fst s) < start))

  300 addAnnotationWorker :: KeywordId -> GHC.SrcSpan -> Delta ()
  301 addAnnotationWorker ann pa = do
  302   if not (isPointSrcSpan pa)
  303     then do
  304       pe <- getPriorEnd
  305       ss <- getSrcSpanDelta
  306       let p = deltaFromSrcSpans pe pa
  307       case (ann,isGoodDelta p) of
  308         (G GHC.AnnComma,False) -> return ()
  309         (G GHC.AnnSemi, False) -> return ()
  310         (G GHC.AnnOpen, False) -> return ()
  311         (G GHC.AnnClose,False) -> return ()
  312         _ -> do
  313           cs <- getUnallocatedComments
  314           let (allocated,cs') = allocatePriorComments cs pa
  315           putUnallocatedComments cs'
  316           return () `debug`("addAnnotationWorker:(ss,pa,allocated,cs)=" ++ showGhc (ss,pa,allocated,cs))
  317           mapM_ addDeltaComment allocated
  318           p' <- adjustDeltaForOffsetM p
  319           addAnnDeltaPos (ss,ann) p'
  320           setPriorEnd pa
  321               -- `debug` ("addDeltaAnnotationWorker:(ss,pe,pa,p,ann)=" ++ show (ss2span ss,ss2span pe,ss2span pa,p,ann))
  322     else do
  323       return ()
  324           -- `debug` ("addDeltaAnnotationWorker::point span:(ss,ma,ann)=" ++ show (ss2span pa,ann))

  328 addDeltaComment :: Comment -> Delta ()
  329 addDeltaComment (Comment paspan str) = do
  330   let pa = span2ss paspan
  331   pe <- getPriorEnd
  332   ss <- getSrcSpanDelta
  333   let p = deltaFromSrcSpans pe pa
  334   p' <- adjustDeltaForOffsetM p
  335   setPriorEnd pa
  336   let e = ss2deltaP (ss2posEnd pe) (snd paspan)
  337   e' <- adjustDeltaForOffsetM e
  338   addAnnDeltaPos (ss,AnnComment (DComment (p',e') str)) p'

  342 -- | Look up and add a Delta annotation at the current position, and
  343 -- advance the position to the end of the annotation
  344 addDeltaAnnotation :: GHC.AnnKeywordId -> Delta ()
  345 addDeltaAnnotation ann = do
  346   ss <- getSrcSpanDelta
  347   when (ann == GHC.AnnVal) (debugM (showGhc ss))
  348   ma <- getAnnotationDelta ann
  349   when (ann == GHC.AnnVal && null ma) (debugM "empty")
  350   case nub ma of -- ++AZ++ TODO: get rid of duplicates earlier
  351     [] -> return () `debug` ("addDeltaAnnotation empty ma for:" ++ show ann)
  352     [pa] -> addAnnotationWorker (G ann) pa
  353     _ -> error $ "addDeltaAnnotation:(ss,ann,ma)=" ++ showGhc (ss,ann,ma)
  354 
  355 -- | Look up and add a Delta annotation appearing beyond the current
  356 -- SrcSpan at the current position, and advance the position to the
  357 -- end of the annotation
  358 addDeltaAnnotationAfter :: GHC.AnnKeywordId -> Delta ()
  359 addDeltaAnnotationAfter ann = do
  360   ss <- getSrcSpanDelta
  361   ma <- getAnnotationDelta ann
  362   let ma' = filter (\s -> not (GHC.isSubspanOf s ss)) ma
  363   case ma' of
  364     [] -> return () `debug` ("addDeltaAnnotation empty ma")
  365     [pa] -> addAnnotationWorker (G ann) pa
  366     _ -> error $ "addDeltaAnnotation:(ss,ann,ma)=" ++ showGhc (ss,ann,ma)
  367 
  368 -- | Look up and add a Delta annotation at the current position, and
  369 -- advance the position to the end of the annotation
  370 addDeltaAnnotationLs :: GHC.AnnKeywordId -> Int -> Delta ()
  371 addDeltaAnnotationLs ann off = do
  372   ma <- getAnnotationDelta ann
  373   case (drop off ma) of
  374     [] -> return ()
  375         -- `debug` ("addDeltaAnnotationLs:missed:(off,pe,ann,ma)=" ++ show (off,ss2span pe,ann,fmap ss2span ma))
  376     (pa:_) -> addAnnotationWorker (G ann) pa
  377 
  378 -- | Look up and add possibly multiple Delta annotation at the current
  379 -- position, and advance the position to the end of the annotations
  380 addDeltaAnnotations :: GHC.AnnKeywordId -> Delta ()
  381 addDeltaAnnotations ann = do
  382   ma <- getAnnotationDelta ann
  383   let do_one ap' = addAnnotationWorker (G ann) ap'
  384                     -- `debug` ("addDeltaAnnotations:do_one:(ap',ann)=" ++ showGhc (ap',ann))
  385   mapM_ do_one (sort ma)
  386 
  387 -- | Look up and add possibly multiple Delta annotations enclosed by
  388 -- the current SrcSpan at the current position, and advance the
  389 -- position to the end of the annotations
  390 addDeltaAnnotationsInside :: GHC.AnnKeywordId -> Delta ()
  391 addDeltaAnnotationsInside ann = do
  392   ss <- getSrcSpanDelta
  393   ma <- getAnnotationDelta ann
  394   let do_one ap' = addAnnotationWorker (G ann) ap'
  395                     -- `debug` ("addDeltaAnnotations:do_one:(ap',ann)=" ++ showGhc (ap',ann))
  396   let filtered = (sort $ filter (\s -> GHC.isSubspanOf s ss) ma)
  397   mapM_ do_one filtered
  398 
  399 -- | Look up and add possibly multiple Delta annotations not enclosed by
  400 -- the current SrcSpan at the current position, and advance the
  401 -- position to the end of the annotations
  402 addDeltaAnnotationsOutside :: GHC.AnnKeywordId -> KeywordId -> Delta ()
  403 addDeltaAnnotationsOutside gann ann = do
  404   ss <- getSrcSpanDelta
  405   if ss2span ss == ((1,1),(1,1))
  406     then return ()
  407     else do
  408       -- ma <- getAnnotationDelta ss gann
  409       ma <- getAndRemoveAnnotationDelta ss gann
  410       let do_one ap' = addAnnotationWorker ann ap'
  411       mapM_ do_one (sort $ filter (\s -> not (GHC.isSubspanOf s ss)) ma)
  412 
  413 -- | Add a Delta annotation at the current position, and advance the
  414 -- position to the end of the annotation
  415 addDeltaAnnotationExt :: GHC.SrcSpan -> GHC.AnnKeywordId -> Delta ()
  416 addDeltaAnnotationExt s ann = do
  417   addAnnotationWorker (G ann) s

  420 addEofAnnotation :: Delta ()
  421 addEofAnnotation = do
  422   pe <- getPriorEnd
  423   ss <- getSrcSpanDelta
  424   ma <- withSrcSpanDelta (GHC.noLoc ()) (DP (0,0)) (getAnnotationDelta GHC.AnnEofPos)
  425   case ma of
  426     [] -> return ()
  427     (pa:pss) -> do
  428       cs <- getUnallocatedComments
  429       mapM_ addDeltaComment cs
  430       let DP (r,c) = deltaFromSrcSpans pe pa
  431       addAnnDeltaPos (ss,G GHC.AnnEofPos) (DP (r, c - 1))
  432       setPriorEnd pa `warn` ("Trailing annotations after Eof: " ++ showGhc pss)
  433 
  434 
  435 countAnnsDelta :: GHC.AnnKeywordId -> Delta Int
  436 countAnnsDelta ann = do
  437   ma <- getAnnotationDelta ann
  438   return (length ma)
