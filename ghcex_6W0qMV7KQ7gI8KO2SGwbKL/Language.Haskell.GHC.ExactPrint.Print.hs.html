   43 -- | Print an AST exactly as specified by the annotations on the nodes in the tree.
   44 -- The output of this function should exactly match the source file.
   45 exactPrint :: Annotate ast => GHC.Located ast -> GHC.ApiAnns -> String
   46 exactPrint ast ghcAnns = exactPrintWithAnns ast relativeAnns
   47   where
   48     relativeAnns = relativiseApiAnns ast ghcAnns
   49 
   50 -- | Print an AST with a map of potential modified `Anns`. The usual way to
   51 -- generate such a map is by calling `relativiseApiAnns`.
   52 exactPrintWithAnns :: Annotate ast
   53                      => GHC.Located ast
   54                      -> Anns
   55                      -> String
   56 exactPrintWithAnns ast an = runEP (markLocated ast) an

   60 -- The EP monad and basic combinators
   61 
   62 data EPState = EPState
   63              { epPos       :: Pos -- ^ Current output position
   64              , epAnns      :: Anns
   65              , epAnnKds    :: [[(KeywordId, DeltaPos)]] -- MP: Could this be moved to the local state with suitable refactoring?
   66                                                         -- AZ, it is already in the last element of Annotation, for withOffset
   67              }
   68 
   69 data EPStack = EPStack
   70              {  epLHS      :: Int -- ^ Marks the column of the LHS of the i
   71                                   --   current layout block
   72              }
   73 
   74 data EPWriter = EPWriter
   75               { output :: Endo String }
   76 
   77 instance Monoid EPWriter where
   78   mempty = EPWriter mempty
   79   (EPWriter a) `mappend` (EPWriter c) = EPWriter (a <> c)
   80 
   81 type EP a = RWS EPStack EPWriter EPState a
   82 
   83 runEP :: Annotated () -> Anns -> String
   84 runEP action ans =
   85   flip appEndo "" . output . snd
   86   . (\next -> execRWS next initialEPStack (defaultEPState ans))
   87   . printInterpret $ action
   88 
   89 
   90 printInterpret :: Annotated a -> EP a
   91 printInterpret = iterTM go
   92   where
   93     go :: AnnotationF (EP a) -> EP a
   94     go (MarkEOF next) =
   95       printStringAtMaybeAnn (G GHC.AnnEofPos) "" >> next
   96     go (MarkPrim kwid mstr next) =
   97       let annString = fromMaybe (keywordToString kwid) mstr in
   98         printStringAtMaybeAnn (G kwid) annString >> next
   99     go (MarkOutside _ kwid next) =
  100       printStringAtMaybeAnnAll kwid ";"  >> next
  101     go (MarkInside akwid next) =
  102       allAnns akwid >> next
  103     go (MarkMany akwid next) =
  104       allAnns akwid >> next
  105     go (MarkOffsetPrim kwid _ mstr next) =
  106       let annString = fromMaybe (keywordToString kwid) mstr in
  107         printStringAtMaybeAnn (G kwid) annString >> next
  108     go (MarkAfter akwid next) =
  109       justOne akwid >> next
  110     go (WithAST lss flag action next) =
  111       exactPC lss flag (NoLayoutRules <$ (printInterpret action)) >> next
  112     go (OutputKD _ next) =
  113       next
  114     go (CountAnns kwid next) =
  115       countAnnsEP (G kwid) >>= next
  116     go (SetLayoutFlag kwid action next) =
  117       setLayout kwid (printInterpret action) >> next
  118     go (MarkExternal _ akwid s next) =
  119       printStringAtMaybeAnn (G akwid) s >> next
  120 
  121 justOne, allAnns :: GHC.AnnKeywordId -> EP ()
  122 justOne kwid = printStringAtMaybeAnn (G kwid) (keywordToString kwid)
  123 allAnns kwid = printStringAtMaybeAnnAll (G kwid) (keywordToString kwid)
  124 
  125 setLayout :: GHC.AnnKeywordId -> EP () -> EP LayoutFlag
  126 setLayout akiwd k = do
  127   p <- gets epPos
  128   local (\s -> s { epLHS = snd p - (length (keywordToString akiwd))})
  129                   (LayoutRules <$ k)

  133 defaultEPState :: Anns -> EPState
  134 defaultEPState as = EPState
  135              { epPos    = (1,1)
  136              , epAnns   = as
  137              , epAnnKds = []
  138              }

  140 initialEPStack :: EPStack
  141 initialEPStack  = EPStack
  142              { epLHS = 0
  143              }

  145 getPos :: EP Pos
  146 getPos = gets epPos

  148 setPos :: Pos -> EP ()
  149 setPos l = modify (\s -> s {epPos = l})

  153 -- | Given an annotation associated with a specific SrcSpan, determines a new offset relative to the previous
  154 -- offset
  155 --
  156 withOffset :: Annotation -> LayoutFlag -> (EP LayoutFlag -> EP LayoutFlag)
  157 withOffset (Ann (DP (edLine, edColumn)) annOffset  _) flag k = do
  158   oldOffset <-  asks epLHS -- Shift from left hand column
  159   (_l, currentColumn) <- getPos
  160   rec
  161     let offset = case (flag <> f) of
  162                        LayoutRules ->  if edLine == 0
  163                         then currentColumn + edColumn
  164                         else oldOffset + annOffset
  165                        NoLayoutRules -> oldOffset
  166     f <-  (local (\s -> s { epLHS = offset }) k)
  167   return f
  168 
  169 -- |Get the current column offset
  170 getOffset :: EP ColOffset
  171 getOffset = asks epLHS

  175 getAndRemoveAnnotation :: (Data a) => GHC.Located a -> EP (Maybe Annotation)
  176 getAndRemoveAnnotation a = do
  177   (r, an') <- gets (getAndRemoveAnnotationEP a . epAnns)
  178   modify (\s -> s { epAnns = an' })
  179   return r
  180 
  181 withKds :: [(KeywordId, DeltaPos)] -> EP a -> EP a
  182 withKds kd action = do
  183   modify (\s -> s { epAnnKds = kd : (epAnnKds s) })
  184   r <- action
  185   modify (\s -> s { epAnnKds = tail (epAnnKds s) })
  186   return r
  187 
  188 -- | Get and remove the first item in the (k,v) list for which the k matches.
  189 -- Return the value, together with any comments skipped over to get there.
  190 destructiveGetFirst :: KeywordId -> ([(KeywordId,v)],[(KeywordId,v)])
  191                     -> ([(KeywordId,v)], Maybe v,[(KeywordId,v)])
  192 destructiveGetFirst _key (acc,[]) = ([], Nothing ,acc)
  193 destructiveGetFirst  key (acc,((k,v):kvs))
  194   | k == key = let (cs,others) = commentsAndOthers acc in (cs, Just v ,others++kvs)
  195   | otherwise = destructiveGetFirst key (acc++[(k,v)],kvs)
  196   where
  197     commentsAndOthers kvs' = partition isComment kvs'
  198     isComment ((AnnComment _),_) = True
  199     isComment _              = False
  200 
  201 -- |destructive get, hence use an annotation once only
  202 getAnnFinal :: KeywordId -> EP ([DComment], Maybe DeltaPos)
  203 getAnnFinal kw = do
  204   kd <- gets epAnnKds
  205   let (r, kd', dcs) = case kd of
  206                   []    -> (Nothing ,[], [])
  207                   (k:kds) -> (r',kk:kds, dcs')
  208                     where (cs', r',kk) = destructiveGetFirst kw ([],k)
  209                           dcs' = concatMap keywordIdToDComment cs'
  210   modify (\s -> s { epAnnKds = kd' })
  211   return (dcs, r)
  212 
  213 -- ---------------------------------------------------------------------

  215 keywordIdToDComment :: (KeywordId, DeltaPos) -> [DComment]
  216 keywordIdToDComment (AnnComment comment,_dp) = [comment]
  217 keywordIdToDComment _                   = []
  218 
  219 -- |non-destructive get
  220 peekAnnFinal :: KeywordId -> EP (Maybe DeltaPos)
  221 peekAnnFinal kw = do
  222   (_, r, _) <- (\kd -> destructiveGetFirst kw ([], kd)) <$> gets (head . epAnnKds)
  223   return r

  227 printString :: String -> EP ()
  228 printString str = do
  229   (l,c) <- gets epPos
  230   setPos (l, c + length str)
  231   tell (mempty {output = Endo $ showString str })
  232 
  233 newLine :: EP ()
  234 newLine = do
  235     (l,_) <- getPos
  236     printString "\n"
  237     setPos (l+1,1)
  238 
  239 padUntil :: Pos -> EP ()
  240 padUntil (l,c) = do
  241     (l1,c1) <- getPos
  242     case  {- trace (show ((l,c), (l1,c1))) -} () of
  243      _ {-()-} | l1 >= l && c1 <= c -> printString $ replicate (c - c1) ' '
  244               | l1 < l             -> newLine >> padUntil (l,c)
  245               | otherwise          -> return ()
  246 
  247 printWhitespace :: Pos -> EP ()
  248 printWhitespace p = do
  249   -- mPrintComments p >> padUntil p
  250   padUntil p
  251 
  252 printStringAt :: Pos -> String -> EP ()
  253 printStringAt p str = printWhitespace p >> printString str

  257 -- |This should be the final point where things are mode concrete,
  258 -- before output. Hence the point where comments can be inserted
  259 printStringAtLsDelta :: [DComment] -> [DeltaPos] -> String -> EP ()
  260 printStringAtLsDelta cs mc s =
  261   case reverse mc of
  262     (cl:_) -> do
  263       p <- getPos
  264       colOffset <- getOffset
  265       if isGoodDeltaWithOffset cl colOffset
  266         then do
  267           mapM_ printQueuedComment cs
  268           printStringAt (undelta p cl colOffset) s
  269             `debug` ("printStringAtLsDelta:(pos,s):" ++ show (undelta p cl colOffset,s))
  270         else return () `debug` ("printStringAtLsDelta:bad delta for (mc,s):" ++ show (mc,s))
  271     _ -> return ()
  272 
  273 
  274 isGoodDeltaWithOffset :: DeltaPos -> Int -> Bool
  275 isGoodDeltaWithOffset dp colOffset = isGoodDelta (DP (undelta (0,0) dp colOffset))
  276 
  277 -- AZ:TODO: harvest the commonality between this and printStringAtLsDelta
  278 printQueuedComment :: DComment -> EP ()
  279 printQueuedComment (DComment (dp,de) s) = do
  280   p <- getPos
  281   colOffset <- getOffset
  282   let (dr,dc) = undelta (0,0) dp colOffset
  283   if isGoodDelta (DP (dr,max 0 dc)) -- do not lose comments against the left margin
  284     then do
  285       printStringAt (undelta p dp colOffset) s
  286          `debug` ("printQueuedComment:(pos,s):" ++ show (undelta p dp colOffset,s))
  287       setPos (undelta p de colOffset)
  288     else return () `debug` ("printQueuedComment::bad delta for (dp,s):" ++ show (dp,s))

  292 printStringAtMaybeAnn :: KeywordId -> String -> EP ()
  293 printStringAtMaybeAnn an str = do
  294   (comments, ma) <- getAnnFinal an
  295   printStringAtLsDelta comments (maybeToList ma) str
  296     -- ++AZ++: Enabling the following line causes a very weird error associated with AnnPackageName. I suspect it is because it is forcing the evaluation of a non-existent an or str
  297     -- `debug` ("printStringAtMaybeAnn:(an,ma,str)=" ++ show (an,ma,str))
  298 
  299 printStringAtMaybeAnnAll :: KeywordId -> String -> EP ()
  300 printStringAtMaybeAnnAll an str = go
  301   where
  302     go = do
  303       (comments, ma) <- getAnnFinal an
  304       case ma of
  305         Nothing -> return ()
  306         Just d  -> printStringAtLsDelta comments [d] str >> go

  310 countAnnsEP :: KeywordId -> EP Int
  311 countAnnsEP an = do
  312   ma <- peekAnnFinal an
  313   return (length ma)

  317 -- |First move to the given location, then call exactP
  318 exactPC :: Data ast => GHC.Located ast -> LayoutFlag -> EP LayoutFlag -> EP LayoutFlag
  319 exactPC ast flag action =
  320     do return () `debug` ("exactPC entered for:" ++ showGhc (GHC.getLoc ast))
  321        ma <- getAndRemoveAnnotation ast
  322        let an@(Ann _ _ kds) = fromMaybe annNone ma
  323        withContext kds an flag action
  324 
  325 withContext :: [(KeywordId, DeltaPos)]
  326             -> Annotation
  327             -> LayoutFlag
  328             -> EP LayoutFlag -> EP LayoutFlag
  329 withContext kds an flag = withKds kds . withOffset an flag
