   42 -- | Print an AST exactly as specified by the annotations on the nodes in the tree.
   43 -- The output of this function should exactly match the source file.
   44 exactPrint :: Annotate ast => GHC.Located ast -> GHC.ApiAnns -> String
   45 exactPrint ast ghcAnns = exactPrintWithAnns ast relativeAnns
   46   where
   47     relativeAnns = relativiseApiAnns ast ghcAnns
   48 
   49 -- | Print an AST with a map of potential modified `Anns`. The usual way to
   50 -- generate such a map is by calling `relativiseApiAnns`.
   51 exactPrintWithAnns :: Annotate ast
   52                      => GHC.Located ast
   53                      -> Anns
   54                      -> String
   55 exactPrintWithAnns ast@(GHC.L l _) an = runEP (markLocated ast) l an

   59 -- The EP monad and basic combinators
   60 
   61 -- The (ColOffset,ColOffset) value carries the normal and current
   62 -- column offset. The second one captures the difference between the
   63 -- original col when the DP was captured and the current one.
   64 
   65 data EPState = EPState
   66              { epPos       :: Pos -- ^ Current output position
   67              , epAnns      :: Anns
   68              , epAnnKds    :: [[(KeywordId, DeltaPos)]] -- MP: Could this be moved to the local state with suitable refactoring?
   69                                                         -- AZ, it is already in the last element of Annotation, for withOffset
   70              }
   71 
   72 data EPStack = EPStack
   73              { epStack     :: (ColOffset,ColDelta) -- ^ stack of offsets that currently apply
   74              , epSrcSpan  :: GHC.SrcSpan
   75              }
   76 
   77 type EP a = RWS EPStack (Endo String) EPState a
   78 
   79 runEP :: Annotated () -> GHC.SrcSpan -> Anns -> String
   80 runEP action ss ans =
   81   flip appEndo "" . snd
   82   . (\next -> execRWS next (initialEPStack ss) (defaultEPState ans))
   83   . printInterpret $ action
   84 
   85 
   86 printInterpret :: Annotated a -> EP a
   87 printInterpret = iterTM go
   88   where
   89     go :: AnnotationF (EP a) -> EP a
   90     go (MarkEOF next) =
   91       printStringAtMaybeAnn (G GHC.AnnEofPos) "" >> next
   92     go (MarkPrim kwid mstr next) =
   93       let annString = fromMaybe (keywordToString kwid) mstr in
   94         printStringAtMaybeAnn (G kwid) annString >> next
   95     go (MarkOutside _ kwid next) =
   96       printStringAtMaybeAnnAll kwid ";"  >> next
   97     go (MarkInside akwid next) =
   98       allAnns akwid >> next
   99     go (MarkMany akwid next) =
  100       allAnns akwid >> next
  101     go (MarkOffsetPrim kwid _ mstr next) =
  102       let annString = fromMaybe (keywordToString kwid) mstr in
  103         printStringAtMaybeAnn (G kwid) annString >> next
  104     go (MarkAfter akwid next) =
  105       justOne akwid >> next
  106     go (WithAST lss _ action next) =
  107       exactPC lss (printInterpret action) >>= next
  108     go (OutputKD _ next) =
  109       next
  110     go (CountAnns kwid next) =
  111       countAnnsEP (G kwid) >>= next
  112     go (SetLayoutFlag next) =
  113       next
  114     go (MarkExternal _ akwid s next) =
  115       printStringAtMaybeAnn (G akwid) s >> next
  116 
  117 justOne, allAnns :: GHC.AnnKeywordId -> EP ()
  118 justOne kwid = printStringAtMaybeAnn (G kwid) (keywordToString kwid)
  119 allAnns kwid = printStringAtMaybeAnnAll (G kwid) (keywordToString kwid)

  122 defaultEPState :: Anns -> EPState
  123 defaultEPState as = EPState
  124              { epPos    = (1,1)
  125              , epAnns   = as
  126              , epAnnKds = []
  127              }

  129 initialEPStack :: GHC.SrcSpan -> EPStack
  130 initialEPStack ss = EPStack
  131              { epStack     = (0,0)
  132              , epSrcSpan   = ss
  133              }

  135 getPos :: EP Pos
  136 getPos = gets epPos

  138 setPos :: Pos -> EP ()
  139 setPos l = modify (\s -> s {epPos = l})

  143 -- | Given an annotation associated with a specific SrcSpan, determines a new offset relative to the previous
  144 -- offset
  145 --
  146 withOffset :: Annotation -> (EP a -> EP a)
  147 withOffset a@(Ann (DP (edLine, edColumn)) newline originalStartCol annDelta _) k = do
  148   -- The colOffset is the offset to be used when going to the next line.
  149   -- The colIndent is how far the current code block has been indented.
  150   (colOffset, colIndent) <- asks epStack
  151   (_l, currentColumn) <- getPos
  152   let
  153       -- Add extra indentation for this SrcSpan
  154      colOffset' = annDelta + colOffset
  155      -- Work out where new column starts, based on the entry delta
  156      newStartColumn = if edLine == 0
  157                           then edColumn + currentColumn -- same line, use entry delta
  158                           else colOffset -- different line, use current offset
  159      newColIndent        = newStartColumn - originalStartCol
  160 
  161      offsetValNewIndent  = (colOffset' + (newColIndent - colIndent), newColIndent)
  162      offsetValSameIndent = (colOffset'            ,                  colIndent)
  163      offsetValNewline    = (annDelta   + colIndent,                  colIndent)
  164 
  165      newOffset =
  166         case newline of
  167           -- For use by AST modifiers, to preserve the indentation
  168           -- level for the next line after an AST modification
  169           KeepOffset  -> offsetValNewline
  170 
  171           -- Generated during the annotation phase
  172           LineChanged       -> offsetValNewline
  173           LayoutLineChanged -> offsetValNewline
  174 
  175           LineSame          -> offsetValSameIndent
  176           LayoutLineSame    -> offsetValNewIndent
  177   local (\s -> s {epStack = newOffset }) k
  178     `debug` ("pushOffset:(a, colOffset, colIndent, currentColumn, newOffset)="
  179                  ++ show (a, colOffset, colIndent, currentColumn, newOffset))
  180 
  181 -- |Get the current column offset
  182 getOffset :: EP ColOffset
  183 getOffset = asks (fst . epStack)

  187 withSrcSpan :: GHC.SrcSpan -> (EP a -> EP a)
  188 withSrcSpan ss = local (\s -> s {epSrcSpan = ss})

  190 getAndRemoveAnnotation :: (Data a) => GHC.Located a -> EP (Maybe Annotation)
  191 getAndRemoveAnnotation a = do
  192   (r, an') <- gets (getAndRemoveAnnotationEP a . epAnns)
  193   modify (\s -> s { epAnns = an' })
  194   return r
  195 
  196 withKds :: [(KeywordId, DeltaPos)] -> EP a -> EP a
  197 withKds kd action = do
  198   modify (\s -> s { epAnnKds = kd : (epAnnKds s) })
  199   r <- action
  200   modify (\s -> s { epAnnKds = tail (epAnnKds s) })
  201   return r
  202 
  203 -- | Get and remove the first item in the (k,v) list for which the k matches.
  204 -- Return the value, together with any comments skipped over to get there.
  205 destructiveGetFirst :: KeywordId -> ([(KeywordId,v)],[(KeywordId,v)])
  206                     -> ([(KeywordId,v)], Maybe v,[(KeywordId,v)])
  207 destructiveGetFirst _key (acc,[]) = ([], Nothing ,acc)
  208 destructiveGetFirst  key (acc,((k,v):kvs))
  209   | k == key = let (cs,others) = commentsAndOthers acc in (cs, Just v ,others++kvs)
  210   | otherwise = destructiveGetFirst key (acc++[(k,v)],kvs)
  211   where
  212     commentsAndOthers kvs' = partition isComment kvs'
  213     isComment ((AnnComment _),_) = True
  214     isComment _              = False
  215 
  216 -- |destructive get, hence use an annotation once only
  217 getAnnFinal :: KeywordId -> EP ([DComment], Maybe DeltaPos)
  218 getAnnFinal kw = do
  219   kd <- gets epAnnKds
  220   let (r, kd', dcs) = case kd of
  221                   []    -> (Nothing ,[], [])
  222                   (k:kds) -> (r',kk:kds, dcs')
  223                     where (cs', r',kk) = destructiveGetFirst kw ([],k)
  224                           dcs' = concatMap keywordIdToDComment cs'
  225   modify (\s -> s { epAnnKds = kd' })
  226   return (dcs, r)
  227 
  228 -- ---------------------------------------------------------------------
  229 
  230 keywordIdToDComment :: (KeywordId, DeltaPos) -> [DComment]
  231 keywordIdToDComment (AnnComment comment,_dp) = [comment]
  232 keywordIdToDComment _                   = []
  233 
  234 -- |non-destructive get
  235 peekAnnFinal :: KeywordId -> EP (Maybe DeltaPos)
  236 peekAnnFinal kw = do
  237   (_, r, _) <- (\kd -> destructiveGetFirst kw ([], kd)) <$> gets (head . epAnnKds)
  238   return r

  242 printString :: String -> EP ()
  243 printString str = do
  244   (l,c) <- gets epPos
  245   setPos (l, c + length str)
  246   tell (Endo $ showString str)
  247 
  248 newLine :: EP ()
  249 newLine = do
  250     (l,_) <- getPos
  251     printString "\n"
  252     setPos (l+1,1)
  253 
  254 padUntil :: Pos -> EP ()
  255 padUntil (l,c) = do
  256     (l1,c1) <- getPos
  257     case  {- trace (show ((l,c), (l1,c1))) -} () of
  258      _ {-()-} | l1 >= l && c1 <= c -> printString $ replicate (c - c1) ' '
  259               | l1 < l             -> newLine >> padUntil (l,c)
  260               | otherwise          -> return ()
  261 
  262 printWhitespace :: Pos -> EP ()
  263 printWhitespace p = do
  264   -- mPrintComments p >> padUntil p
  265   padUntil p
  266 
  267 printStringAt :: Pos -> String -> EP ()
  268 printStringAt p str = printWhitespace p >> printString str

  272 -- |This should be the final point where things are mode concrete,
  273 -- before output. Hence the point where comments can be inserted
  274 printStringAtLsDelta :: [DComment] -> [DeltaPos] -> String -> EP ()
  275 printStringAtLsDelta cs mc s =
  276   case reverse mc of
  277     (cl:_) -> do
  278       p <- getPos
  279       colOffset <- getOffset
  280       if isGoodDeltaWithOffset cl colOffset
  281         then do
  282           mapM_ printQueuedComment cs
  283           printStringAt (undelta p cl colOffset) s
  284             `debug` ("printStringAtLsDelta:(pos,s):" ++ show (undelta p cl colOffset,s))
  285         else return () `debug` ("printStringAtLsDelta:bad delta for (mc,s):" ++ show (mc,s))
  286     _ -> return ()
  287 
  288 
  289 isGoodDeltaWithOffset :: DeltaPos -> Int -> Bool
  290 isGoodDeltaWithOffset dp colOffset = isGoodDelta (DP (undelta (0,0) dp colOffset))
  291 
  292 -- AZ:TODO: harvest the commonality between this and printStringAtLsDelta
  293 printQueuedComment :: DComment -> EP ()
  294 printQueuedComment (DComment (dp,de) s) = do
  295   p <- getPos
  296   colOffset <- getOffset
  297   let (dr,dc) = undelta (0,0) dp colOffset
  298   if isGoodDelta (DP (dr,max 0 dc)) -- do not lose comments against the left margin
  299     then do
  300       printStringAt (undelta p dp colOffset) s
  301          `debug` ("printQueuedComment:(pos,s):" ++ show (undelta p dp colOffset,s))
  302       setPos (undelta p de colOffset)
  303     else return () `debug` ("printQueuedComment::bad delta for (dp,s):" ++ show (dp,s))

  307 printStringAtMaybeAnn :: KeywordId -> String -> EP ()
  308 printStringAtMaybeAnn an str = do
  309   (comments, ma) <- getAnnFinal an
  310   printStringAtLsDelta comments (maybeToList ma) str
  311     -- ++AZ++: Enabling the following line causes a very weird error associated with AnnPackageName. I suspect it is because it is forcing the evaluation of a non-existent an or str
  312     -- `debug` ("printStringAtMaybeAnn:(an,ma,str)=" ++ show (an,ma,str))
  313 
  314 printStringAtMaybeAnnAll :: KeywordId -> String -> EP ()
  315 printStringAtMaybeAnnAll an str = go
  316   where
  317     go = do
  318       (comments, ma) <- getAnnFinal an
  319       case ma of
  320         Nothing -> return ()
  321         Just d  -> printStringAtLsDelta comments [d] str >> go

  325 countAnnsEP :: KeywordId -> EP Int
  326 countAnnsEP an = do
  327   ma <- peekAnnFinal an
  328   return (length ma)

  332 -- |First move to the given location, then call exactP
  333 exactPC :: Data ast => GHC.Located ast -> EP a -> EP a
  334 exactPC a@(GHC.L l _ast) action =
  335     do return () `debug` ("exactPC entered for:" ++ showGhc l)
  336        ma <- getAndRemoveAnnotation a
  337        let an@(Ann _edp _nl _sc _dc kds) = fromMaybe annNone ma
  338        withContext kds l an action
  339 
  340 withContext :: [(KeywordId, DeltaPos)]
  341             -> GHC.SrcSpan
  342             -> Annotation
  343             -> EP a -> EP a
  344 withContext kds l an = withKds kds . withSrcSpan l . withOffset an
