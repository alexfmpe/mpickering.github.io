   42 -- | A Haskell comment.
   43 data Comment = Comment Span String
   44   deriving (Eq,Show,Typeable,Data)
   45 
   46 instance GHC.Outputable Comment where
   47   ppr x = GHC.text (show x)
   48 
   49 -- |Delta version of the comment.
   50 data DComment = DComment (DeltaPos,DeltaPos) String
   51   deriving (Eq,Show,Typeable,Data,Ord)
   52 
   53 instance Ord Comment where
   54   compare (Comment p1 _) (Comment p2 _) = compare p1 p2
   55 
   56 type PosToken = (GHC.Located GHC.Token, String)
   57 
   58 type Pos = (Int,Int)
   59 type Span = (Pos,Pos)
   60 
   61 data DeltaPos = DP (Int,Int) deriving (Show,Eq,Ord,Typeable,Data)
   62 type ColOffset = Int -- ^ indentation point for a new line
   63 type ColDelta  = Int -- ^ difference between two cols
   64 type Col       = Int
   65 
   66 annNone :: Annotation
   67 annNone = Ann (DP (0,0)) LineSame 0 0 []
   68 
   69 combineAnns :: Annotation -> Annotation -> Annotation
   70 combineAnns (Ann ed1 nl1 c1 dp1 dps1) (Ann _ed2 _nl2 _c2 _dp2 dps2)
   71   = Ann ed1 nl1 c1 dp1 (dps1 ++ dps2)
   72 
   73 data LineChanged = LineSame | LineChanged
   74                  | KeepOffset -- ^ For use in AST editing
   75                  | LayoutLineSame | LayoutLineChanged -- experimental, may replace LineSame and LineChanged
   76                  deriving (Show,Eq,Typeable)
   77 
   78 data Annotation = Ann
   79   {
   80     ann_entry_delta  :: !DeltaPos -- ^ Offset used to get to the start
   81                                   -- of the SrcSpan, during the
   82                                   -- annotatePC phase
   83   , ann_original_nl  :: !LineChanged -- ^ Did the original span start
   84                                      -- on a new line wrt to the prior
   85                                      -- one?
   86   , ann_original_col :: !Col      -- ^ Start of the SrcSpan, as used
   87                                   -- during the annotatePC phase
   88   , ann_delta        :: !ColOffset -- ^ Indentation level introduced
   89                                    -- by this SrcSpan, for other items
   90                                    -- at same layout level
   91   , anns             :: [(KeywordId, DeltaPos)] -- TODO:AZ change this to ann_dps
   92 
   93   } deriving (Typeable,Eq)
   94 
   95 instance Show Annotation where
   96   show (Ann dp nl c d ans) = "(Ann (" ++ show dp ++ ") " ++ show nl ++ " " ++ show c ++ " " ++ show d ++ " " ++ show ans ++ ")"
   97 
   98 instance Monoid Annotation where
   99   mempty = annNone
  100   mappend = combineAnns
  101 
  102 type Anns = Map.Map AnnKey Annotation
  103 
  104 -- | For every @Located a@, use the @SrcSpan@ and constructor name of
  105 -- a as the key, to store the standard annotation.
  106 -- These are used to maintain context in the AP and EP monads
  107 data AnnKey   = AnnKey GHC.SrcSpan AnnConName
  108                   deriving (Eq, Show, Ord)
  109 
  110 mkAnnKey :: (Data a) => GHC.Located a -> AnnKey
  111 mkAnnKey (GHC.L l a) = AnnKey l (annGetConstr a)
  112 
  113 -- Holds the name of a constructor
  114 data AnnConName = CN String
  115                  deriving (Eq,Show,Ord)
  116 
  117 annGetConstr :: (Data a) => a -> AnnConName
  118 annGetConstr a = CN (show $ toConstr a)
  119 
  120 unConName :: AnnConName -> String
  121 unConName (CN s) = s
  122 
  123 -- |We need our own version of keywordid to distinguish between a
  124 -- semi-colon appearing within an AST element and one separating AST
  125 -- elements in a list.
  126 data KeywordId = G GHC.AnnKeywordId
  127                | AnnSemiSep
  128                | AnnComment DComment
  129                | AnnList GHC.SrcSpan -- ^ In some circumstances we
  130                                      -- need to annotate a list of
  131                                      -- statements (e.g. HsDo) and
  132                                      -- must synthesise a SrcSpan to
  133                                      -- hang the annotations off. This
  134                                      -- needs to be preserved so that
  135                                      -- exactPC can find it, after
  136                                      -- potential AST edits.
  137                deriving (Eq,Show,Ord)
  138 
  139 data LayoutFlag = LayoutRules | NoLayoutRules deriving (Show, Eq)
  140 
  141 instance Monoid LayoutFlag where
  142   mempty = NoLayoutRules
  143   LayoutRules `mappend` _ = LayoutRules
  144   _ `mappend` LayoutRules = LayoutRules
  145   _ `mappend` _           = NoLayoutRules

  166 -- ResTyGADT has a SrcSpan for the original sigtype, we need to create
  167 -- a type for exactPC and annotatePC
  168 data ResTyGADTHook name = ResTyGADTHook [GHC.LHsTyVarBndr name]
  169                    deriving (Typeable)
  170 deriving instance (GHC.DataId name) => Data (ResTyGADTHook name)
  171 deriving instance (Show (GHC.LHsTyVarBndr name)) => Show (ResTyGADTHook name)
  172 
  173 instance (GHC.OutputableBndr name) => GHC.Outputable (ResTyGADTHook name) where
  174   ppr (ResTyGADTHook bs) = GHC.text "ResTyGADTHook" GHC.<+> GHC.ppr bs

  178 getAnnotationEP :: (Data a) =>  GHC.Located a -> Anns -> Maybe Annotation
  179 getAnnotationEP  (GHC.L ss a) annotations =
  180   Map.lookup (AnnKey ss (annGetConstr a)) annotations
  181 
  182 getAndRemoveAnnotationEP :: (Data a)
  183                          => GHC.Located a -> Anns -> (Maybe Annotation,Anns)
  184 getAndRemoveAnnotationEP (GHC.L ss a) annotations
  185  = let key = AnnKey ss (annGetConstr a) in
  186     case Map.lookup key annotations of
  187          Nothing  -> (Nothing, annotations)
  188          Just av -> (Just av, Map.delete key annotations)
