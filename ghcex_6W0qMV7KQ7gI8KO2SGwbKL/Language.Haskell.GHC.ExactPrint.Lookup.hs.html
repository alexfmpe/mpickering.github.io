   18       AnnCloseS  -> "]"
   19       AnnColon    -> ":"
   20       AnnComma   -> ","

   22       AnnDarrow  -> "=>"
   23       AnnData   -> "data"
   24       AnnDcolon  -> "::"
   25       AnnDefault    -> "default"
   26       AnnDeriving   -> "deriving"
   27       AnnDo   -> "do"
   28       AnnDot   -> "."
   29       AnnDotdot  -> ".."

   31       AnnEqual    -> "="
   32       AnnExport   -> "export"
   33       AnnFamily   -> "family"
   34       AnnForall   -> "forall"
   35       AnnForeign    -> "foreign"
   36 --      AnnFunId   -> ""
   37       AnnGroup    -> "group"
   38 --      AnnHeader  -> ""
   39 --      for CType
   40       AnnHiding   -> "hiding"
   41       AnnIf   -> "if"
   42       AnnImport   -> "import"
   43       AnnIn   -> "in"
   44 --      AnnInfix   -> ""
   45       AnnInstance   -> "instance"
   46       AnnLam    -> "\\"

   49       AnnMdo    -> "mdo"
   50       AnnMinus   -> "-"

   68       AnnThen   -> "then"

   71       AnnType   -> "type"

   81 
   82       x -> error $ "keywordToString: missing case for:" ++ show x
