   42 data AnnotationF next where
   43   MarkEOF  :: next -> AnnotationF next
   44   MarkPrim :: GHC.AnnKeywordId -> Maybe String -> next -> AnnotationF next
   45   MarkExternal :: GHC.SrcSpan -> GHC.AnnKeywordId -> String -> next -> AnnotationF next
   46   MarkOutside :: GHC.AnnKeywordId -> KeywordId -> next -> AnnotationF next
   47   MarkInside :: GHC.AnnKeywordId -> next -> AnnotationF next
   48   MarkMany :: GHC.AnnKeywordId -> next -> AnnotationF next
   49   MarkOffsetPrim :: GHC.AnnKeywordId -> Int -> Maybe String -> next -> AnnotationF next
   50   MarkAfter :: GHC.AnnKeywordId -> next -> AnnotationF next
   51   WithAST  :: Data a => GHC.Located a -> LayoutFlag -> Annotated b -> (b -> next) -> AnnotationF next
   52   CountAnns ::  GHC.AnnKeywordId -> (Int -> next) -> AnnotationF next
   53   -- Abstraction breakers
   54   SetLayoutFlag ::  next -> AnnotationF next
   55   OutputKD :: (DeltaPos, (GHC.SrcSpan, KeywordId)) -> next -> AnnotationF next
   56 
   57 deriving instance Functor (AnnotationF)
   58

   78 mark :: GHC.AnnKeywordId -> Annotated ()
   79 mark kwid = markPrim kwid Nothing
   80 
   81 markWithString :: GHC.AnnKeywordId -> String -> Annotated ()
   82 markWithString kwid s = markPrim kwid (Just s)
   83 
   84 markOffsetWithString :: GHC.AnnKeywordId -> Int -> String -> Annotated ()
   85 markOffsetWithString kwid n s = markOffsetPrim kwid n (Just s)
   86 
   87 markOffset :: GHC.AnnKeywordId -> Int -> Annotated ()
   88 markOffset kwid n = markOffsetPrim kwid n Nothing

   90 withAST :: Data a => GHC.Located a -> LayoutFlag -> Annotated b -> Annotated b
   91 withAST lss layout action = liftF (WithAST lss layout prog id)
   92   where
   93     prog = do
   94       r <- action
   95       -- Automatically add any trailing comma or semi
   96       markAfter GHC.AnnComma
   97       markOutside GHC.AnnSemi AnnSemiSep
   98       return r

  102 -- | Constructs a syntax tree which contains information about which
  103 -- annotations are required by each element.
  104 markLocated :: (Annotate ast) => GHC.Located ast -> Annotated ()
  105 markLocated a = withLocated a NoLayoutRules markAST
  106 
  107 withLocated :: Data a
  108             => GHC.Located a
  109             -> LayoutFlag
  110             -> (GHC.SrcSpan -> a -> Annotated ())
  111             -> Annotated ()
  112 withLocated a@(GHC.L l ast) layoutFlag action =
  113   withAST a layoutFlag (action l ast)
  114 
  115 markMaybe :: (Annotate ast) => Maybe (GHC.Located ast) -> Annotated ()
  116 markMaybe Nothing    = return ()
  117 markMaybe (Just ast) = markLocated ast
  118 
  119 markList :: (Annotate ast) => [GHC.Located ast] -> Annotated ()
  120 markList xs = mapM_ markLocated xs
  121 
  122 -- | Flag the item to be annotated as requiring layout.
  123 markWithLayout :: Annotate ast => GHC.Located ast -> Annotated ()
  124 markWithLayout a = withLocated a LayoutRules markAST
  125 
  126 markListWithLayout :: Annotate [GHC.Located ast] => GHC.SrcSpan -> [GHC.Located ast] -> Annotated ()
  127 markListWithLayout l ls = do
  128   let ss = getListSrcSpan ls
  129   outputKD $ ((DP (0,0)), (l,AnnList ss))
  130   markWithLayout (GHC.L ss ls)
  131 
  132 -- ---------------------------------------------------------------------
  133 -- Managing lists which have been separated, e.g. Sigs and Binds
  134 
  135 prepareListAnnotation :: Annotate a => [GHC.Located a] -> [(GHC.SrcSpan,Annotated ())]
  136 prepareListAnnotation ls = map (\b@(GHC.L l _) -> (l,markLocated b)) ls
  137 
  138 applyListAnnotations :: [(GHC.SrcSpan, Annotated ())] -> Annotated ()
  139 applyListAnnotations ls
  140   = mapM_ snd $ sortBy (\(a,_) (b,_) -> compare a b) ls

  149 instance Annotate (GHC.HsModule GHC.RdrName) where
  150   markAST _ (GHC.HsModule mmn mexp imps decs mdepr _haddock) = do
  151 
  152 
  153     case mmn of
  154       Nothing -> return ()
  155       Just (GHC.L ln mn) -> do
  156         mark GHC.AnnModule
  157         markExternal ln GHC.AnnVal (GHC.moduleNameString mn)
  158 
  159     case mdepr of
  160       Nothing -> return ()
  161       Just depr -> markLocated depr
  162 
  163     case mexp of
  164       Nothing   -> return ()
  165       Just expr -> markLocated expr
  166 
  167     mark GHC.AnnWhere
  168     mark GHC.AnnOpenC -- Possible '{'
  169     markMany GHC.AnnSemi -- possible leading semis
  170     mapM_ markLocated imps
  171 
  172     markList decs
  173 
  174     mark GHC.AnnCloseC -- Possible '}'
  175 
  176     markEOF

  238 instance Annotate GHC.RdrName where
  239   markAST l n = do
  240     case rdrName2String n of
  241       "[]" -> do
  242         mark GHC.AnnOpenS  -- '['
  243         mark GHC.AnnCloseS -- ']'
  244       "()" -> do
  245         mark GHC.AnnOpenP  -- '('
  246         mark GHC.AnnCloseP -- ')'
  247       "(##)" -> do
  248         markWithString GHC.AnnOpen  "(#" -- '(#'
  249         markWithString GHC.AnnClose  "#)"-- '#)'
  250       "[::]" -> do
  251         markWithString GHC.AnnOpen  "[:" -- '[:'
  252         markWithString GHC.AnnClose ":]" -- ':]'
  253       str ->  do
  254         mark GHC.AnnType
  255         mark GHC.AnnOpenP -- '('
  256         markOffset GHC.AnnBackquote 0
  257         markMany GHC.AnnCommaTuple -- For '(,,,)'
  258         cnt <- countAnns GHC.AnnVal
  259         cntT <- countAnns GHC.AnnCommaTuple
  260         cntR <- countAnns GHC.AnnRarrow
  261         case cnt of
  262           0 -> if cntT >0 || cntR >0
  263                  then return ()
  264                  else markExternal l GHC.AnnVal str
  265           1 -> markWithString GHC.AnnVal str
  266           x -> error $ "markP.RdrName: too many AnnVal :" ++ showGhc (l,x)
  267         mark GHC.AnnTildehsh
  268         mark GHC.AnnTilde
  269         mark GHC.AnnRarrow
  270         markOffset GHC.AnnBackquote 1
  271         mark GHC.AnnCloseP -- ')'

  282 instance (GHC.DataId name,Annotate name)
  283   => Annotate (GHC.ImportDecl name) where
  284  markAST _ imp@(GHC.ImportDecl msrc (GHC.L ln _) _pkg src safeflag _qual _impl _as hiding) = do
  285 
  286    -- 'import' maybe_src maybe_safe optqualified maybe_pkg modid maybeas maybeimpspec
  287    mark GHC.AnnImport
  288 
  289    -- "{-# SOURCE" and "#-}"
  290    when src (markWithString GHC.AnnOpen (fromMaybe "{-# SOURCE" msrc)
  291              >> markWithString GHC.AnnClose "#-}")
  292    when safeflag (mark GHC.AnnSafe)
  293    mark GHC.AnnQualified
  294    mark GHC.AnnPackageName
  295 
  296    markExternal ln GHC.AnnVal (GHC.moduleNameString $ GHC.unLoc $ GHC.ideclName imp)
  297 
  298    case GHC.ideclAs imp of
  299       Nothing -> return ()
  300       Just mn -> do
  301           mark GHC.AnnAs
  302           markWithString GHC.AnnVal (GHC.moduleNameString mn)
  303 
  304    case hiding of
  305      Nothing -> return ()
  306      Just (_isHiding,lie) -> do
  307        mark GHC.AnnHiding
  308        markLocated lie

 1253   = case spans of
 1254       []  -> GHC.noSrcSpan
 1255       sss -> GHC.combineSrcSpans (head sss) (last sss)
 1256   where
 1257     spans = sort (map GHC.getLoc (GHC.bagToList binds) ++ map GHC.getLoc sigs)
