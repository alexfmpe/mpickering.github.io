import Data.Function
import System.IO
import Test.QuickCheck
import Common
import Data.Functor.Identity

import "logic-TPTP" Codec.TPTP

main ::  IO ()
main = quickCheckWith (stdArgs { maxSuccess = 5000 }) prop_test_ie
