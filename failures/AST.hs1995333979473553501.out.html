import Data.Monoid
import Data.Vector (Vector)
import qualified Data.Vector as V

data Json = Number Double
          | String Text
          | Bool Bool
          | Null
          | Array (Vector Json)
          | Object (HashMap Text Json)
          deriving (Eq)
