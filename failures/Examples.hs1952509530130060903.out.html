example_json = "{\"name\": \"jon\", \"job\":\"Code\", \"age\":42, \"things\":[1,2,3] }"

nested_json = "{\"dad\" : { \"name\" : \"bob\" } }"

parsed_example :: Maybe (PlainRec '[("name" ::: Text),
